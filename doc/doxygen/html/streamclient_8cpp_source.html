239  void stream_client_socket::shutdown(int method)
240  {
241  int u_method = 0; // unix flags
242 
243  // Already shut down using this method...
244  if ( (method & (LIBSOCKET_READ|LIBSOCKET_WRITE)) && (shut_rd == true) && (shut_wr == true) )
245  return;
246  if ( (method & LIBSOCKET_READ) && (shut_rd == true) )
247  return;
248  if ( (method & LIBSOCKET_WRITE) && (shut_wr == true) )
249  return;
250 
251 #if LIBSOCKET_LINUX || BD_ANDROID
252  using BERKELEY::SHUT_RDWR;
253  using BERKELEY::SHUT_RD;
254  using BERKELEY::SHUT_WR;
255 #endif
256 
257  if ( method == (LIBSOCKET_READ|LIBSOCKET_WRITE) )
258  u_method = SHUT_RDWR;
259  else if ( method == LIBSOCKET_READ )
260  u_method = SHUT_RD;
261  else if ( method == LIBSOCKET_WRITE )
262  u_method = SHUT_WR;
263  else // With no valid combination
264  return;
265 
266  if ( 0 > BERKELEY::shutdown(sfd,u_method)) // It's equal whether we use this or its brother from libunixsocket
267  {
268  throw socket_exception(__FILE__,__LINE__,"stream_client_socket::shutdown() - Could not shutdown socket");
269  }
270 
271  if ( method & LIBSOCKET_READ )
272  shut_rd = true;
273  if ( method & LIBSOCKET_WRITE )
274  shut_wr = true;
275  }
