135  {
136  struct epoll_event new_event;
137 
138  new_event.data.ptr = 0; // ptr is the largest field (8 bytes on 64bit)
139  new_event.events = 0;
140 
141  if ( method & LIBSOCKET_READ )
142  new_event.events |= EPOLLIN;
143  if ( method & LIBSOCKET_WRITE )
144  new_event.events |= EPOLLOUT;
145 
146  new_event.data.ptr = &sock;
147 
148  if ( 0 > epoll_ctl(epollfd,EPOLL_CTL_ADD,sock.getfd(),&new_event) )
149  throw socket_exception(__FILE__,__LINE__,string("epoll_ctl failed: ") + strerror(errno));
